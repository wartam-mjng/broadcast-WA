Panduan Google Apps Script

Gunakan skrip di bawah ini agar integrasi database & WhatsApp lancar

Pastikan fungsi doPost(e) di Google Apps Script Anda memproses data payload dengan aman seperti skrip di bawah ini agar kompatibel dengan GitHub Pages:

function doPost(e) {
  try {
    // 1. Parsing payload dari berbagai jenis format request (JSON / Form-URL / PlainText)
    var postData;
    if (e.parameter && e.parameter.payload) {
      postData = JSON.parse(e.parameter.payload);
    } else if (e.postData && e.postData.contents) {
      postData = JSON.parse(e.postData.contents);
    } else {
      throw new Error("Tidak ada data payload yang dikirim.");
    }

    // 2. Baca isi parameter yang dikirim oleh sistem frontend
    var action = postData.action;
    var pengirim = postData.pengirim;
    var pesanTemplate = postData.pesan;
    var kontakList = postData.kontak; // Array kontak yang dipilih

    if (action === 'sendMessage') {
      var berhasil = 0;
      var gagal = 0;

      // Jalankan perulangan kirim pesan ke masing-masing nomor
      for (var i = 0; i < kontakList.length; i++) {
        var k = kontakList[i];
        
        // Ganti variabel nama {nama} di pesan
        var pesanPersonal = pesanTemplate.replace(/{nama}/g, k.nama);
        
        // EKSEKUSI KIRIM KE GATEWAY WHATSAPP ANDA (Contoh: Fonnte)
        var response = kirimKeGatewayWhatsApp(k.nomor, pesanPersonal);
        
        if (response.success) {
          berhasil++;
          // Tulis log berhasil ke Google Sheet Anda di sini jika diperlukan
        } else {
          gagal++;
        }
      }

      // Kembalikan respon sukses yang bisa dibaca oleh halaman web Anda
      return ContentService.createTextOutput(JSON.stringify({
        status: "success",
        berhasil: berhasil,
        gagal: gagal
      })).setMimeType(ContentService.MimeType.JSON);
    }
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({
      status: "error",
      message: error.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

// Fungsi pembantu untuk mengirim data ke gateway WA Anda
function kirimKeGatewayWhatsApp(nomor, pesan) {
  // Ganti URL dan Token sesuai spesifikasi Gateway yang Anda sewa
  var url = "https://api.fonnte.com/send"; 
  var token = "TOKEN_API_WHATSAPP_ANDA_DISINI"; 

  var payload = {
    target: nomor,
    message: pesan
  };

  var options = {
    method: "POST",
    headers: {
      "Authorization": token
    },
    payload: payload,
    muteHttpExceptions: true
  };

  try {
    var res = UrlFetchApp.fetch(url, options);
    var resJson = JSON.parse(res.getContentText());
    
    // Sesuaikan kondisi sukses berdasarkan respon dari Gateway Anda
    if (resJson.status === true || resJson.status === "pending" || res.getResponseCode() == 200) {
      return { success: true };
    }
    return { success: false };
  } catch (e) {
    return { success: false };
  }
}
